const WebSocket = require('ws'); const AuthHandler = require('./Network/AuthHandler'); const CharacterHandler = require('./Network/CharacterHandler'); // Nouvel import const wss = new WebSocket.Server({ port: 8080 }); console.log("Serveur Tactical RPG démarré sur le port 8080..."); wss.on('connection', function connection(ws) { console.log("Nouveau client connecté."); ws.on('message', function incoming(message) { try { const packet = JSON.parse(message); // Routage des requêtes du client if (packet.action === "register") { AuthHandler.handleRegister(ws, packet.data); } else if (packet.action === "login") { AuthHandler.handleLogin(ws, packet.data); } else if (packet.action === "create_character") { // On passe le packet complet pour que le Handler puisse lire le Token CharacterHandler.handleCreateCharacter(ws, packet); } else { console.log(`Action non reconnue : ${packet.action}`); } } catch (e) { console.error("Format de paquet JSON invalide :", e); } }); ws.on('close', () => { console.log("Client déconnecté."); }); });
